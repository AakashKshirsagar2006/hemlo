#include <iostream>
#include <string>

using namespace std;

int main() {
    string d, g, t, r;

    // 1. Inputs
    cout << "Enter Data & Generator: ";
    cin >> d >> g;

    int dl = d.length(), gl = g.length();
    t = d + string(gl - 1, '0'); // Automatically append N-1 zeros

    // 2. Sender: CRC Generation using the XOR trick
    for(int i = 0; i < dl; i++)
        if(t[i] == '1')
            for(int j = 0; j < gl; j++)
                t[i+j] = (t[i+j] ^ g[j]) + '0'; // THE MAGIC TRICK

    cout << "Transmitted Message: " << d << t.substr(dl) << "\n\n";

    // 3. Receiver: Error Check
    cout << "Enter Received Message: ";
    cin >> r;

    for(int i = 0; i < r.length() - gl + 1; i++)
        if(r[i] == '1')
            for(int j = 0; j < gl; j++)
                r[i+j] = (r[i+j] ^ g[j]) + '0';

    // 4. Verdict
    bool err = false;
    for(int i = r.length() - gl + 1; i < r.length(); i++)
        if(r[i] == '1') err = true;

    cout << (err ? "Result: Error detected!\n" : "Result: No error.\n");

    return 0;
}

void xor(char *a, char *b, char *result) {
    for (int i = 1; i < strlen(b); i++) {
        result[i - 1] = (a[i] == b[i]) ? '0' : '1';
    }
    result[strlen(b) - 1] = '\0';
}

void crc(char *data, char *divisor) {
    char temp[100], result[100];
    int n = strlen(divisor);

    strncpy(temp, data, n);

    for (int i = n; i <= strlen(data); i++) {
        if (temp[0] == '1')
            xor(temp, divisor, result);
        else
            xor(temp, "0000", result); // same length as divisor

        if (i < strlen(data)) {
            strncat(result, &data[i], 1);
        }
        strcpy(temp, result);
    }

    printf("CRC Remainder: %s\n", result);
}

int main() {
    char data[100], divisor[10];

    printf("Enter data: ");

    printf("Enter divisor: ");

    // Append zeros
    int d = strlen(divisor);
    for (int i = 0; i < d - 1; i++) {
        strcat(data, "0");
    }
